				OnActCodeTypes=t_OnTakingDamageEn,t_OnTakingDamageSp,t_OnStep,t_OnTakingDamageMi

                OnActCode ^{
				    01=dword domik=ScriptItemActShip();
					02=int damage=0;
					03=dword thisInfo=CurInfo();

					    01=if(ScriptItemActionType(t_OnTakingDamageEn) || ScriptItemActionType(t_OnTakingDamageSp) || ScriptItemActionType(t_OnTakingDamageMi))
