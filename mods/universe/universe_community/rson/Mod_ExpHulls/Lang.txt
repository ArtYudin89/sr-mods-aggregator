            01=if(!cur_ship) exit;

            09=ReleaseItemFromScript(item);
            10=FreeItem(GetItemFromStorage(FindItemInStorage(item)));
            11=AddItemToStorage(cur_place, GetItemFromVault(text));
        }
        05=else
        05 ^{
            01=if(ItemExtraSpecialsCountByType(ScriptItemActObject1(), FindBonusByNameInCfg('ExpHulls_HullTransporterMark'))) exit;

            03=if(!cur_place) cur_place = GetShipRuins(cur_ship);
