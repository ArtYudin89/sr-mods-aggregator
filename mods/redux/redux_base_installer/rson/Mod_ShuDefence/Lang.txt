ShipInfo ^{
    AddInfo ^{
        CustomInfos ^{
            Absorber ^{
                Description=Система противодействия баллистическому огню, способная нейтрализовать 80% урона от попадания ядерной боеголовки или любой иной ракеты со схожим тротиловым эквивалентом, рассеивая локализованную энергию в вакууме.                 Icon=550
                Name=Абсорбер
                OnActCode ^{
                    01=int charge=0;
                    02=dword info=CurInfo();
                    03=dword ship=ScriptItemActShip();
                    04=int i=ShipCustomShipInfoData(0,info,1);
                    05=if(ShipOwner(ship)==5 && !ShipSubType(ship)) charge=4;
                    06=else charge=1;
                    07=if(ScriptItemActionType(t_OnStep,11) && i<30) ShipCustomShipInfoData(0,info,1,i+charge);
                    08=if(i<30) exit;
                    09=if(i>30) ShipCustomShipInfoData(0,info,1,30);
                    10=if(ScriptItemActionType(t_OnTakingDamageMi))
                    11 ^{
                        01=if(ScriptItemActParam()<120) exit;
                        02=else
                        03 ^{
                            01=ScriptItemActParam(ScriptItemActParam()/5);
                            02=ShipCustomShipInfoData(0,info,1,0);
                        }
                    }
                }
                OnActCodeTypes=t_OnStep,t_OnTakingDamageMi
                OnActStepTypes=11
            }
            ECM ^{
                Description=Мощное устройство, рассеивающее тепловой эффект от любого повреждения обшивке корабля с вероятностью 50%. При рассеивании теряется заряд, что довольно логично.                 Name=Точечный усилитель поля

                    }
                }

            }
