    ShuDomiksPlus_MinigunShot ^{
        00=[t_OnWeaponShot|]

        03=dword ship = ScriptItemActShip();

        08=for(i=15;ShipSpecialBonuses(Player(),bonWSplinter)>i;i=i+15)
        09 ^{
            01=dmg = dmg + 1;
        }
        10=if(ShipOwner(target) != Kling) dmg = dmg * 2;
        11=ScriptItemActParam(dmg);
