ActCodeBlocksList ^{
    Cat_Corvette_OnActCode ^{
        00=[t_OnStep|0,1,3,5,7,9,11]
        01=if(ScriptItemActionType(t_OnStep, 0))
        02 ^{
            01=int pro_in_hull=ItemExtraSpecialsCountByType(ShipEqInSlot(ScriptItemActShip(), t_Hull),FindBonusByNameInCfg('RefArtsUpgrade_PRO3'));
        }
        03=if(pro_in_hull<1)exit;
        04=dword ship=ScriptItemActShip(), msl=0, hit_msl=0, star=ShipStar(ship);
        05=if(!ShipInNormalSpace(ship))exit;
        06=if(StarMissiles(star)==0)exit;
        07=int a=0, min_dist=1000, msl_dist=0;
        08=for(a=0;a<StarMissiles(star);a=a+1)
        09 ^{
            01=msl=StarMissiles(star, a);
            02=msl_dist=Dist(ship, msl);
            03=if(msl_dist>range)continue;
            04=if(MissileTarget(msl)!=ship)continue;
            05=if(MissileOwner(msl)==ship)continue;
            06=if(msl_dist>min_dist)continue;
            07=min_dist=msl_dist;
            08=hit_msl=msl;
        }
        10=if(!hit_msl)exit;
        11=ShowEffect('Weapon.PDTurret', 0, hit_msl, ship);
        12=MissileLive(hit_msl, 1201);
    }
}
